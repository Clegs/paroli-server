Get the next line from the console and call 'callback' with the line as the argument.

 
getLine = (callback) ->
  process.stdin.resume()
  process.stdin.setEncoding 'utf8'
  process.stdin.once 'data', (line) ->
    process.stdin.pause()
    line = line.trim()
    callback(line)
